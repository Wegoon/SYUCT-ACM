二分

  • 确定一个区间，二分每次将不符合性质的一半砍掉
  • 二分其实就是找端点，要么是前段的后端点，要么是后段的前端点
  • 整数模板记模板二选一记下来，实数二分就给了一个模板，没得选


整数二分模板一：

情况1：要找的答案为前段的后端点
int bm(int l,int r)
{
	int ans=l;
	while(l<=r){
		int mid=(l+r)/2;//数学写法
		//int mid=l+r>>1;//二进制写法
		if(check(mid)){
			ans=mid;//mid可能是答案，更新ans
			r=mid-1;
		}else{
			l=mid+1;
		}
	}
	return ans;
}
情况2：要找的答案为后段的前端点

{
	while(l<r){
		int mid=(l+r)/2;//数学写法
		//int mid=l+r>>1;//二进制写法
		if(check(mid)){
			r=mid;
		}else{
			l=mid+1;
		}
	}
	return l;
	//return r;//l==r,返回l或r都行
}
情况2：要找的答案为后段的前端点
int bm(int l,int r)
{
	while(l<r){
		int mid=(l+r+1)/2;//数学写法
		//int mid=l+r+1>>1;//二进制写法
		if(check(mid)){
			l=mid;
		}else{
			r=mid-1;
		}
	}
	return l;
	//return r;//l==r,返回l或r都行
}
